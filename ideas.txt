MAX_TRAINING_FRAMES = 256
MIN_BATCH_SIZE = 4


def get_gear_data(gear: int) -> tuple[int, int]:
    if gear < 1:
        raise ValueError("gear must be >= 1")

    seq_len = (gear * 2) + 2
    batch_size = max(1, MAX_TRAINING_FRAMES // seq_len)
    return seq_len, batch_size


def get_max_gear() -> int:
    gear = 1
    while True:
        _, batch_size = get_gear_data(gear)
        if batch_size < MIN_BATCH_SIZE:
            return gear - 1
        gear += 1


def get_valid_gears() -> range:
    return range(1, get_max_gear() + 1)
